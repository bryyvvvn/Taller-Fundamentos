PORTAL resta(DIAMANTE a, DIAMANTE b) {
    DIAMANTE r; r = a - b; LETRERO(r); TESORO r;
}
PORTAL mult(DIAMANTE a, DIAMANTE b) {
    DIAMANTE r; r = a * b; LETRERO(r); TESORO r;
}
PORTAL divi(DIAMANTE a, DIAMANTE b) {
    DIAMANTE r; r = a / b; LETRERO(r); TESORO r;
}

// Exponente con ZOMBIE
PORTAL exponente(DIAMANTE base, DIAMANTE exp) {
    DIAMANTE r; r = 1;
    DIAMANTE c; c = exp;
    ZOMBIE (c > 0) {
        r = r * base;
        c = c - 1;
    }
    LETRERO(r);
    TESORO r;
}
